95 LoadResult loadStream(std::istream& in) const {
96 LoadResult result;
97
98 std::string line;
99 std::unordered_map<std::string, size_t> header_index;
100
101 if (options_.has_header && std::getline(in, line)) {
102 auto header_fields = parseCsvLine(line);
103 for (size_t i = 0; i < header_fields.size(); ++i) {
104 header_index[normalizeHeader(header_fields[i])] = i;
105 }
106 }
107
108 size_t row = 1;
109 while (std::getline(in, line)) {
110 ++row;
111 if (line.empty()) {
112 continue;
113 }
114
115 auto fields = parseCsvLine(line);
116 auto stock_opt = parseStock(fields, header_index);
117 if (!stock_opt.has_value()) {
118 result.errors.push_back("Row " + std::to_string(row) + ": malformed or missing required fields; skipped.");
119 continue;
120 }
121
122 result.stocks.push_back(*stock_opt);
123 }
124
125 return result;
126 }

138 std::vector<std::string> parseCsvLine(const std::string& line) const {
139 std::vector<std::string> fields;
140 std::string field;
141 bool in_quotes = false;
142
143 for (size_t i = 0; i < line.size(); ++i) {
144 char c = line[i];
145 if (c == '"') {
146 if (in_quotes && i + 1 < line.size() && line[i + 1] == '"') {
147 field.push_back('"');
148 ++i;
149 } else {
150 in_quotes = !in_quotes;
151 }
152 } else if (c == options_.delimiter && !in_quotes) {
153 fields.push_back(trim(field));
154 field.clear();
155 } else {
156 field.push_back(c);
157 }
158 }
159
160 fields.push_back(trim(field));
161 return fields;
162 }

170 std::string normalizeHeader(const std::string& header) const {
171 std::string out;
172 out.reserve(header.size());
173 for (char c : header) {
174 if (std::isalnum(static_cast<unsigned char>(c))) {
175 out.push_back(static_cast<char>(std::tolower(static_cast<unsigned char>(c))));
176 }
177 }
178 return out;
179 }

187 std::string trim(const std::string& value) const {
188 size_t start = 0;
189 while (start < value.size() && std::isspace(static_cast<unsigned char>(value[start]))) {
190 ++start;
191 }
192 size_t end = value.size();
193 while (end > start && std::isspace(static_cast<unsigned char>(value[end - 1]))) {
194 --end;
195 }
196 return value.substr(start, end - start);
197 }
