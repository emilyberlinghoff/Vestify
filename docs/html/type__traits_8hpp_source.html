337template<typename T>
338struct is_range
339{
340 private:
341 using t_ref = typename std::add_lvalue_reference<T>::type;
342
343 using iterator = detected_t<result_of_begin, t_ref>;
344 using sentinel = detected_t<result_of_end, t_ref>;
345
346 // to be 100% correct, it should use https://en.cppreference.com/w/cpp/iterator/input_or_output_iterator
347 // and https://en.cppreference.com/w/cpp/iterator/sentinel_for
348 // but reimplementing these would be too much work, as a lot of other concepts are used underneath
349 static constexpr auto is_iterator_begin =
350 is_iterator_traits<iterator_traits<iterator>>::value;
351
352 public:
353 static constexpr bool value = !std::is_same<iterator, nonesuch>::value && !std::is_same<sentinel, nonesuch>::value && is_iterator_begin;
354};

624// We exclude json_pointer here, because the checks using Compare(A, B) will
625// use json_pointer::operator string_t() which triggers a deprecation warning

627// is_json_pointer_of can be removed once the deprecated function has been

630struct is_comparable < Compare, A, B, enable_if_t < !is_json_pointer_of<A, B>::value
