196 ScoredStock scoreStock(const Stock& stock) const {
197 const ScoringStrategy& strategy = getActiveStrategy();
198
199 ScoredStock result;
200 result.stock = stock;
201 double weighted_sum = 0.0;
202 double total_weight = 0.0;
203
204 for (const auto& [model_name, weight] : strategy.weights) {
205 auto model_it = models_.find(model_name);
206 if (model_it == models_.end()) {
207 std::cerr << "[StrategyManager] warning: model \""
208 << model_name << "\" not registered, skipping.\n";
209 continue;
210 }
211
212 double sub_score = model_it->second->calculateScore(stock);
213 if (!std::isfinite(sub_score)) {
214 sub_score = 0.0;
215 }
216 sub_score = std::clamp(sub_score, 0.0, 100.0);
217 result.subscores[model_name] = sub_score;
218
219 weighted_sum += weight * sub_score;
220 total_weight += weight;
221 }
222
223 if (total_weight > 0.0) {
224 result.composite = weighted_sum / total_weight;
225 }
226 return result;
227 }

235 std::vector<ScoredStock> rankStocks(const std::vector<Stock>& stocks) const {
236 std::vector<ScoredStock> scored;
237 scored.reserve(stocks.size());
238
239 for (const auto& s : stocks) {
240 scored.push_back(scoreStock(s));
241 }
242
243 std::sort(scored.begin(), scored.end(),
244 [](const ScoredStock& a, const ScoredStock& b) {
245 return a.composite > b.composite;
246 });
247
248 return scored;
249 }
